- Switch VLAN and trunk port setup:

enable
configure terminal
vlan 10
name Management
exit

vlan 20
name Kali
exit

vlan 30
name Windows
exit

interface Gi0/0
switchport trunk encapsulation dot1q
switchport mode trunk
exit

end
write memory

- Assigning VMs to VLANS:

enable
configure terminal
interface Gi0/1
switchport mode access
switchport access vlan 10
spanning-tree portfast
exit

interface Gi0/2
switchport mode access
switchport access vlan 10
spanning-tree portfast
exit

interface Gi0/3
switchport mode access
switchport access vlan 20
spanning-tree portfast
exit

interface Gi1/0
switchport mode access
switchport access vlan 30
spanning-tree portfast
exit

end
write memory

- SPAN port setup:

enable
configure terminal
monitor session 1 source interface Gi0/0
monitor session 1 destination interface Gi2/2

interface Gi2/2
spanning-tree portfast
exit

end
write memory
